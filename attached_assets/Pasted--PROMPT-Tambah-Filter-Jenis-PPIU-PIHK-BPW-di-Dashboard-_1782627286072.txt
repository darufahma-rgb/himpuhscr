## LANGKAH 2 — Tambah fungsi rebuildJenisFilter

Di dalam `<script>`, cari fungsi `rebuildKotaFilter()`. Tambahkan fungsi baru tepat setelahnya:

```js
function rebuildJenisFilter() {
  const sel = document.getElementById('jenisFilter');
  const cur = sel.value;
  // Ambil jenis unik dari data, sederhanakan label
  const jenisMap = {};
  DATA.forEach(t => {
    const raw = (t.jenis_anggota || '').trim();
    if (!raw) return;
    // Sederhanakan: ambil kata kunci utama
    let label = raw;
    if (raw.includes('Haji Khusus')) label = 'PIHK — Penyelenggara Haji Khusus';
    else if (raw.includes('Ibadah Umrah') || raw.includes('Perjalanan Ibadah Umrah')) label = 'PPIU — Penyelenggara Umrah';
    else if (raw.includes('Biro Perjalanan')) label = 'BPW — Biro Perjalanan Wisata';
    jenisMap[label] = raw; // label → nilai asli
  });
  // Rebuild options
  const current = sel.value;
  sel.innerHTML = '<option value="">Semua Jenis</option>';
  Object.keys(jenisMap).sort().forEach(label => {
    const opt = document.createElement('option');
    opt.value = label;
    opt.textContent = label;
    sel.appendChild(opt);
  });
  sel.value = current || '';
}
```

## LANGKAH 3 — Update fungsi applyFilter

Cari fungsi `applyFilter()`. Di dalam filter DATA, tambahkan kondisi jenis:

```js
function applyFilter() {
  const q = document.getElementById('searchInput').value.toLowerCase();
  const kota = document.getElementById('kotaFilter').value;
  const status = document.getElementById('statusFilter').value;
  const jenis = document.getElementById('jenisFilter').value; // ← tambah

  filtered = DATA.filter(t => {
    const matchQ = !q || 
      t.nama_perusahaan.toLowerCase().includes(q) || 
      (t.merek_dagang||'').toLowerCase().includes(q);
    const matchKota = !kota || t.kota === kota;
    const matchStatus = !status || getStatus(t.id) === status;
    
    // Filter jenis: cocokkan dengan label yang disederhanakan
    let matchJenis = true;
    if (jenis) {
      const raw = (t.jenis_anggota || '').trim();
      if (jenis.includes('PIHK')) matchJenis = raw.includes('Haji Khusus');
      else if (jenis.includes('PPIU')) matchJenis = raw.includes('Umrah');
      else if (jenis.includes('BPW')) matchJenis = raw.includes('Biro Perjalanan');
      else matchJenis = raw === jenis;
    }
    
    return matchQ && matchKota && matchStatus && matchJenis;
  });

  document.getElementById('counter').textContent = 
    `Menampilkan ${filtered.length} dari ${DATA.length} travel`;
  renderTable();
  renderStats();
}
```

## LANGKAH 4 — Panggil rebuildJenisFilter saat data dimuat

Cari baris `rebuildKotaFilter()` yang dipanggil setelah data dimuat (di dalam `loadData`).
Tambahkan `rebuildJenisFilter()` tepat setelahnya:

```js
if (isNew) {
  rebuildKotaFilter();
  rebuildJenisFilter(); // ← tambah
}
```
